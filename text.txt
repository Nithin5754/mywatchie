         const btns = document.querySelectorAll('.add-btn');

          const closeButtons = document.querySelectorAll('.close-button');

          btns.forEach(btn => {
            btn.addEventListener('click', () => {
              btn.style.display="none"
              const boxId = btn.getAttribute('data-box');

              const box = document.getElementById(boxId);

              if (box) {
                box.style.display = 'block';
              }
            });
          });

          closeButtons.forEach(closebtn => {
            closebtn.addEventListener('click', () => {
              btns.forEach((btn)=>btn.style.display="block")
              const close = closebtn.closest('.open-qty');
             
              if (close) {
                close.style.display = 'none';
              }
            });
          });

    </script>

      function completePurchase() {
        // Check the selected payment method
        const paymentMethod = document.querySelector('input[name="paymentMethod"]:checked').value;

        if (paymentMethod === 'cash') {
          // Redirect to the order success page
          window.location.href = '/confirmOrder';
        } else {
           window.location.href = '/orderManagement';
        }
      }



















           const placeOrder=document.querySelector('#place-order-button');
     
     placeOrder.addEventListener('click',()=>{
       const selectedPayment = document.querySelector('input[name="paymentMethod"]:checked')
       
       if(selectedPayment){
        const paymentMethod=selectedPayment.value 
        if(paymentMethod){

               fetch('/validatePayment', {
            method: 'POST',
            headers: {
              'Content-Type': 'application/json',
            },
            body: JSON.stringify({ paymentMethod }),
          })
           .then(response => response.json())
              .then(data => {
                   // Handle redirection based on the payment method
                   if (data.success) {
                     if (paymentMethod === 'cash') {
                       window.location.href = '/confirmOrder';
                     }
                   } else {
                        window.location.href = '/orderManagement';
                   }
                 })
                       .catch(error => {
                   console.error('Error:', error);
                 });

        }
       }
       
    })
